52 template <typename T1, typename T2, typename T3>
53 inline auto unblocked_cholesky_lambda(T1& L_A, T2& L, T3& A) {
54  return [L_A, L, A]() mutable {
55  const size_t N = A.rows();
56  // Algorithm is in rowmajor so we make the adjoint copy rowmajor
57  Eigen::Matrix<double, -1, -1, Eigen::RowMajor> adjL(L.rows(), L.cols());
58  adjL.template triangularView<Eigen::Lower>() = L.adj();
59  for (int i = N - 1; i >= 0; --i) {
60  for (int j = i; j >= 0; --j) {
61  if (i == j) {
62  A.adj()(i, j) = 0.5 * adjL.coeff(i, j) / L_A.coeff(i, j);
63  } else {
64  A.adj()(i, j) = adjL.coeff(i, j) / L_A.coeff(j, j);
65  adjL.coeffRef(j, j)
66  -= adjL.coeff(i, j) * L_A.coeff(i, j) / L_A.coeff(j, j);
67  }
68  for (int k = j - 1; k >= 0; --k) {
69  adjL.coeffRef(i, k) -= A.adj().coeff(i, j) * L_A.coeff(j, k);
70  adjL.coeffRef(j, k) -= A.adj().coeff(i, j) * L_A.coeff(i, k);
71  }
72  }
73  }
74  };
75 }

83 template <typename T1, typename T2, typename T3>
84 inline auto cholesky_lambda(T1& L_A, T2& L, T3& A) {
85  return [L_A, L, A]() mutable {
86  using Block_ = Eigen::Block<Eigen::MatrixXd>;
87  using Eigen::Lower;
88  using Eigen::StrictlyUpper;
89  using Eigen::Upper;
90  Eigen::MatrixXd L_adj(L.rows(), L.cols());
91  L_adj.template triangularView<Eigen::Lower>() = L.adj();
92  const int M_ = L_A.rows();
93  int block_size_ = std::max(M_ / 8, 8);
94  block_size_ = std::min(block_size_, 128);
95  for (int k = M_; k > 0; k -= block_size_) {
96  int j = std::max(0, k - block_size_);
97  auto R = L_A.block(j, 0, k - j, j);
98  auto D = L_A.block(j, j, k - j, k - j).eval();
99  auto B = L_A.block(k, 0, M_ - k, j);
100  auto C = L_A.block(k, j, M_ - k, k - j);
101  auto R_adj = L_adj.block(j, 0, k - j, j);
102  auto D_adj = L_adj.block(j, j, k - j, k - j);
103  auto B_adj = L_adj.block(k, 0, M_ - k, j);
104  auto C_adj = L_adj.block(k, j, M_ - k, k - j);
105  D.transposeInPlace();
106  if (C_adj.size() > 0) {
107  C_adj = D.template triangularView<Upper>()
108  .solve(C_adj.transpose())
109  .transpose();
110  B_adj.noalias() -= C_adj * R;
111  D_adj.noalias() -= C_adj.transpose() * C;
112  }
113  D_adj = (D * D_adj.template triangularView<Lower>()).eval();
114  D_adj.template triangularView<StrictlyUpper>()
115  = D_adj.adjoint().template triangularView<StrictlyUpper>();
116  D.template triangularView<Upper>().solveInPlace(D_adj);
117  D.template triangularView<Upper>().solveInPlace(D_adj.transpose());
118  R_adj.noalias() -= C_adj.transpose() * B;
119  R_adj.noalias() -= D_adj.template selfadjointView<Lower>() * R;
120  D_adj.diagonal() *= 0.5;
121  }
122  A.adj().template triangularView<Eigen::Lower>() = L_adj;
123  };
124 }

135 inline auto opencl_cholesky_lambda(AMat& arena_A, LVari& vari_L) {
136  return [arena_A, vari_L]() mutable {
137  const int M_ = arena_A.rows();
138  const int packed_size = M_ * (M_ + 1) / 2;
139  Eigen::Map<Eigen::Matrix<vari*, Eigen::Dynamic, 1>> L_cpu(
140  vari_L, M_ * (M_ + 1) / 2);
141  Eigen::VectorXd L_val_cpu = L_cpu.val();
142  Eigen::VectorXd L_adj_cpu = L_cpu.adj();
143  matrix_cl<double> L_val = packed_copy<matrix_cl_view::Lower>(L_val_cpu, M_);
144  matrix_cl<double> L_adj = packed_copy<matrix_cl_view::Lower>(L_adj_cpu, M_);
145  int block_size
146  = M_
147  / std::max(1,
148  opencl_context.tuning_opts().cholesky_rev_block_partition);
149  block_size = std::max(block_size, 8);
150  block_size = std::min(
151  block_size, opencl_context.tuning_opts().cholesky_rev_min_block_size);
152  // The following is an OpenCL implementation of
153  // the chain() function from the cholesky_block
154  // vari class implementation
155  for (int k = M_; k > 0; k -= block_size) {
156  const int j = std::max(0, k - block_size);
157  const int k_j_ind = k - j;
158  const int m_k_ind = M_ - k;
159 
160  auto&& R_val = block_zero_based(L_val, j, 0, k_j_ind, j);
161  auto&& R_adj = block_zero_based(L_adj, j, 0, k_j_ind, j);
162  matrix_cl<double> D_val = block_zero_based(L_val, j, j, k_j_ind, k_j_ind);
163  matrix_cl<double> D_adj = block_zero_based(L_adj, j, j, k_j_ind, k_j_ind);
164  auto&& B_val = block_zero_based(L_val, k, 0, m_k_ind, j);
165  auto&& B_adj = block_zero_based(L_adj, k, 0, m_k_ind, j);
166  auto&& C_val = block_zero_based(L_val, k, j, m_k_ind, k_j_ind);
167  auto&& C_adj = block_zero_based(L_adj, k, j, m_k_ind, k_j_ind);
168 
169  C_adj = C_adj * tri_inverse(D_val);
170  B_adj = B_adj - C_adj * R_val;
171  D_adj = D_adj - transpose(C_adj) * C_val;
172 
173  D_adj = transpose(D_val) * D_adj;
174  D_adj.triangular_transpose<TriangularMapCL::LowerToUpper>();
175  D_val = transpose(tri_inverse(D_val));
176  D_adj = D_val * transpose(D_val * D_adj);
177  D_adj.triangular_transpose<TriangularMapCL::LowerToUpper>();
178 
179  R_adj = R_adj - transpose(C_adj) * B_val - D_adj * R_val;
180  diagonal(D_adj) = diagonal(D_adj) * 0.5;
181 
182  block_zero_based(L_adj, j, j, k_j_ind, k_j_ind) = D_adj;
183  }
184  L_adj.view(matrix_cl_view::Lower);
185  std::vector<double> L_adj_cpu_res = packed_copy(L_adj);
186  int pos = 0;
187  for (Eigen::Index j = 0; j < arena_A.rows(); ++j) {
188  for (Eigen::Index i = j; i < arena_A.rows(); ++i) {
189  arena_A.coeffRef(i, j)->adj_ += L_adj_cpu_res[pos];
190  pos++;
191  }
192  }
193  };
194 }

211 inline auto cholesky_decompose(const EigMat& A) {
212  check_square("cholesky_decompose", "A", A);
213  arena_t<EigMat> arena_A = A;
214  arena_t<Eigen::Matrix<double, -1, -1>> L_A(arena_A.val());
215 #ifdef STAN_OPENCL
216  L_A = cholesky_decompose(L_A);
217 #else
218  check_symmetric("cholesky_decompose", "A", A);
219  Eigen::LLT<Eigen::Ref<Eigen::MatrixXd>, Eigen::Lower> L_factor(L_A);
220  check_pos_definite("cholesky_decompose", "m", L_factor);
221 #endif
222  L_A.template triangularView<Eigen::StrictlyUpper>().setZero();
223  // looping gradient calcs faster for small matrices compared to
224  // cholesky_block
225  vari* dummy = new vari(0.0, false);
226  arena_t<EigMat> L(L_A.rows(), L_A.cols());
227  if (L_A.rows() <= 35) {
228  internal::initialize_return(L, L_A, dummy);
229  reverse_pass_callback(internal::unblocked_cholesky_lambda(L_A, L, arena_A));
230  } else {
231 #ifdef STAN_OPENCL

233  > opencl_context.tuning_opts().cholesky_size_worth_transfer) {
234  vari** vari_L = ChainableStack::instance_->memalloc_.alloc_array<vari*>(
235  arena_A.rows() * (arena_A.rows() + 1) / 2);
236  size_t pos = 0;
237  for (Eigen::Index j = 0; j < arena_A.rows(); ++j) {
238  for (Eigen::Index i = j; i < arena_A.rows(); ++i) {
239  vari_L[pos] = new vari(L_A.coeffRef(i, j), false);
240  L.coeffRef(i, j).vi_ = vari_L[pos];
241  ++pos;
242  }

244  L.coeffRef(k, j).vi_ = dummy;
245  }
246  }

248  } else {

250  reverse_pass_callback(internal::cholesky_lambda(L_A, L, arena_A));
251  }
252 #else
253  internal::initialize_return(L, L_A, dummy);
254  reverse_pass_callback(internal::cholesky_lambda(L_A, L, arena_A));
